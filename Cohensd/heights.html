computeD <- function(pop1,pop2) {
  sd1 <- sd(pop1)
  sd2 <- sd(pop2)
  pooledSD <- ((length(pop1)-1)*sd1+(length(pop2)-1)*sd2)/(length(pop1)+length(pop2)-2)
  d <- (mean(pop1)-mean(pop2))/pooledSD
  d
}
